data = np.linspace(0, 2*n*np.pi, 300)
# n * 2pi
def sinusoidal(sty):
    plt.style.use(sty)
    fig, ax = plt.subplots()
    ax.plot(data, np.sin(data), label='Sine')
    ax.plot(data, np.cos(data), label='Cosine')
    ax.legend()

sinusoidal('default')
